076        try {
077          SimpleDateFormat sdf = new SimpleDateFormat("MM/dd/yyyy");
078          Date d = sdf.parse(pString);
079          cal = Calendar.getInstance();
080          cal.setTime(d);
081        }
082        catch(ParseException e) {   }
083      }
